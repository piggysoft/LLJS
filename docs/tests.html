/* Copy parameters into their stack locations if their address
 * is ever taken. */

function test_0 (int x) {
  let int *y = &x;
  test(0, *y, x);
  test(1, *y, 1);
}

test_0(1);

/* Cast function expression to dyn type. */

let foo = function int bar(int x, int y) {
  return x + y;
};

test(0, foo("4", 2.1), 42);
